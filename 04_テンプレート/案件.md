## サブタスク一覧

```dataviewjs
const fmt = d => d?.toFormat ? d.toFormat("yyyy-MM-dd") : (d ?? "-");
const folder = dv.current().file.folder;
const moc = dv.current().file.name;
const tasks = dv.pages(`"${folder}"`).where(p => p.file.name !== moc);
dv.table(["サブタスク", "担当", "ステータス", "期限", "進捗"],
    tasks.sort(p => p.ステータス).map(p => [
        p.file.link, p.担当, p.ステータス, fmt(p.期限),
        (p.工数進捗 ?? "-") + (p.工数進捗 != null ? "%" : "")
    ])
);
```

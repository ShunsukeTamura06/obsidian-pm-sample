## 進行中サブタスク

```dataviewjs
const fmt = d => d?.toFormat ? d.toFormat("yyyy-MM-dd") : (d ?? "-");
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const tasks = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件" && !isMOC(p) && p.ステータス === "進行中")
    .sort(p => p.期限);
dv.table(["サブタスク", "案件", "担当", "期限", "進捗"],
    tasks.map(t => [
        t.file.link,
        t.file.folder.split("/").pop(),
        t.担当, fmt(t.期限),
        (t.工数進捗 ?? "-") + (t.工数進捗 != null ? "%" : "")
    ])
);
```

## 要対応（期限14日以内）

```dataviewjs
const fmt = d => d?.toFormat ? d.toFormat("yyyy-MM-dd") : (d ?? "-");
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const pages = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件"
        && p.ステータス !== "完了"
        && p.期限
        && dv.date(p.期限) <= dv.date("today").plus({days: 14}))
    .sort(p => p.期限);
dv.table(["ノート", "種別", "担当", "期限"],
    pages.map(p => [p.file.link, isMOC(p) ? "案件" : "サブタスク", p.担当, fmt(p.期限)])
);
```

## 担当者別ロード

```dataviewjs
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const tasks = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件" && !isMOC(p) && p.ステータス !== "完了");
const byPerson = {};
for (const t of tasks) {
    const name = t.担当 || "未割当";
    if (!byPerson[name]) byPerson[name] = {total:0, inp:0};
    byPerson[name].total++;
    if (t.ステータス === "進行中") byPerson[name].inp++;
}
dv.table(["担当者", "引き受けサブタスク", "うち進行中"],
    Object.entries(byPerson).sort((a,b)=>b[1].total-a[1].total).map(([n,v])=>[n,v.total,v.inp])
);
```

const isMOC = p => p.file.folder.split("/").pop() === p.file.name;

## 進行中タスク

```dataviewjs
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const tasks = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件" && !isMOC(p) && p.ステータス === "進行中")
    .sort(p => p.期限);
dv.table(["タスク", "案件", "担当", "期限", "進捗"],
    tasks.map(t => [
        t.file.link,
        t.file.folder.split("/").pop(),
        t.担当, t.期限,
        (t.工数進捗 ?? "-") + (t.工数進捗 != null ? "%" : "")
    ])
);
```

## 要対応（期限14日以内）

```dataviewjs
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const pages = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件"
        && p.ステータス !== "完了"
        && p.期限
        && dv.date(p.期限) <= dv.date("today").plus({days: 14}))
    .sort(p => p.期限);
dv.table(["ノート", "種別", "担当", "期限"],
    pages.map(p => [p.file.link, isMOC(p) ? "案件" : "タスク", p.担当, p.期限])
);
```

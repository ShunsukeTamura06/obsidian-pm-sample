# 全案件一覧

## 案件一覧

```dataview
TABLE 担当, ステータス, 優先度, 期限
FROM "01_案件"
WHERE 種別 = "案件"
SORT 優先度 DESC, 期限 ASC
```

## 案件別タスク集計

```dataviewjs
const projects = dv.pages('"01_案件"')
    .where(p => p.種別 === "案件")
    .sort(p => p.file.name);

const rows = [];
for (const p of projects) {
    const tasks = dv.pages(`"01_案件/${p.file.name}"`)
        .where(t => t.種別 === "タスク");
    const total = tasks.length;
    if (total === 0) {
        rows.push([p.file.link, 0, "-", "-", "-"]);
        continue;
    }
    const done = tasks.filter(t => t.ステータス === "完了").length;
    const inp  = tasks.filter(t => t.ステータス === "進行中").length;
    const ns   = tasks.filter(t => t.ステータス === "未着手").length;
    const avg  = Math.round(tasks.array().reduce((s, t) => s + (t.工数進捗 || 0), 0) / total);
    rows.push([p.file.link, total, done, inp, ns, avg + "%"]);
}
dv.table(["案件", "タスク数", "完了", "進行中", "未着手", "平均進捗"], rows);
```

## 要対応（期限14日以内かつ未完了）

```dataview
TABLE file.link AS ノート, 種別, 担当, 期限
FROM "01_案件"
WHERE (種別 = "案件" OR 種別 = "タスク")
    AND ステータス != "完了"
    AND date(期限) <= date(today) + dur(14 days)
SORT 期限 ASC
```

# ガントチャート

## タスクガント

```dataviewjs
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;
const tasks = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件" && !isMOC(p) && p.開始日 && p.期限)
    .sort(p => p.開始日);

const statusMap = {"完了":"done", "進行中":"active", "保留":"crit", "未着手":""};
let g = "gantt\n    title タスクガント\n    dateFormat YYYY-MM-DD\n    axisFormat %m/%d\n";

const grouped = {};
for (const t of tasks) {
    const key = t.file.folder.split("/").pop();
    if (!grouped[key]) grouped[key] = [];
    grouped[key].push(t);
}

for (const [sec, items] of Object.entries(grouped)) {
    g += `    section ${sec}\n`;
    for (const t of items) {
        const st = statusMap[t.ステータス] || "";
        const s = t.開始日.toISODate ? t.開始日.toISODate() : t.開始日;
        const e = t.期限.toISODate ? t.期限.toISODate() : t.期限;
        g += `    ${t.file.name} :${st ? st + ", " : ""}${s}, ${e}\n`;
    }
}
await dv.paragraph("```mermaid\n" + g + "\n```");
```

> **凡例**: 青＝進行中 / 緑＝完了 / 赤＝保留 / グレー＝未着手

## 案件別タスク一覧

```dataviewjs
const isMOC = p => p.file.folder.split("/").pop() === p.file.name;

// 案件（MOC）を取得
const projects = dv.pages('"01_案件"')
    .where(p => p.file.folder !== "01_案件" && isMOC(p))
    .sort(p => p.file.name);

for (const proj of projects) {
    // 案件名を見出しとして表示
    dv.header(3, proj.file.link);

    // その案件フォルダのタスクを取得
    const tasks = dv.pages(`"${proj.file.folder}"`)
        .where(p => p.file.name !== proj.file.name)
        .sort(p => p.開始日);

    if (tasks.length === 0) {
        dv.paragraph("タスクノートなし");
        continue;
    }

    dv.table(
        ["タスク", "担当", "ステータス", "期限", "進捗", "概要"],
        tasks.map(t => [
            t.file.link,
            t.担当 ?? "-",
            t.ステータス ?? "-",
            t.期限 ?? "-",
            t.工数進捗 != null ? t.工数進捗 + "%" : "-",
            t.概要 ?? "-"
        ])
    );
}
```

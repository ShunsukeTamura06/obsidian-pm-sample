# ガントチャート

## 全案件ガント（親プロジェクト別）

```dataviewjs
const pages = dv.pages('"01_案件"')
    .where(p => p.開始日 && p.期限 && p["案件レベル"] !== "親")
    .sort(p => p.開始日);
const statusMap = {"完了":"done","進行中":"active","保留":"crit","レビュー中":"active","未着手":""};
let g = "gantt\n    title 案件ガント\n    dateFormat YYYY-MM-DD\n    axisFormat %m/%d\n";
const grouped = {};
for (const p of pages) {
    let key = p["案件レベル"] === "子" && p["親案件"]
        ? "▼ " + p["親案件"].path.split("/").pop().replace(".md","")
        : p["案件種別"] || "その他";
    if (!grouped[key]) grouped[key] = [];
    grouped[key].push(p);
}
const parentSec = Object.keys(grouped).filter(k=>k.startsWith("▼")).sort();
const otherSec = Object.keys(grouped).filter(k=>!k.startsWith("▼")).sort();
for (const sec of [...parentSec, ...otherSec]) {
    g += `    section ${sec}\n`;
    for (const p of grouped[sec]) {
        const st = statusMap[p["ステータス"]] || "";
        const s = p.開始日.toISODate ? p.開始日.toISODate() : p.開始日;
        const e = p.期限.toISODate ? p.期限.toISODate() : p.期限;
        g += `    ${p.file.name.replace(/:/g,"_")} :${st?st+", ":""}${s}, ${e}\n`;
    }
}
await dv.paragraph("```mermaid\n" + g + "\n```");
```

## 担当者別ガント

```dataviewjs
const pages = dv.pages('"01_案件"')
    .where(p => p.開始日 && p.期限 && p["ステータス"] !== "完了" && p["案件レベル"] !== "親")
    .sort(p => p.開始日);
const statusMap = {"進行中":"active","保留":"crit","レビュー中":"active","未着手":""};
let g = "gantt\n    title 担当者別ガント\n    dateFormat YYYY-MM-DD\n    axisFormat %m/%d\n";
const grouped = {};
for (const p of pages) {
    const key = p["担当"] || "未割当";
    if (!grouped[key]) grouped[key] = [];
    grouped[key].push(p);
}
for (const [sec, items] of Object.entries(grouped)) {
    g += `    section ${sec}\n`;
    for (const p of items) {
        const st = statusMap[p["ステータス"]] || "";
        const s = p.開始日.toISODate ? p.開始日.toISODate() : p.開始日;
        const e = p.期限.toISODate ? p.期限.toISODate() : p.期限;
        g += `    ${p.file.name.replace(/:/g,"_")} :${st?st+", ":""}${s}, ${e}\n`;
    }
}
await dv.paragraph("```mermaid\n" + g + "\n```");
```

> 凡例: グレー=未着手 / 青=進行中・レビュー中 / 緑=完了 / 赤=保留
